import math

    def resolver_segundo_grau():
    print("--- Calculadora de Equação de 2º Grau ---")
    
    # Entrada de dados
    a = float(input("Digite o valor de a: "))
    
    if a == 0:
        print("O valor de 'a' não pode ser zero em uma equação de 2º grau.")
        return

    b = float(input("Digite o valor de b: "))
    c = float(input("Digite o valor de c: "))

    # Cálculo do Delta (Δ = b² - 4ac)
    delta = b**2 - 4*a*c
    print(f"\nDelta (Δ) = {delta}")

    if delta < 0:
        print("A equação não possui raízes reais (Delta negativo).")
    elif delta == 0:
        x = -b / (2 * a)
        print(f"A equação possui uma única raiz real: x = {x}")
    else:
        # Bhaskara: x = (-b ± √Δ) / 2a
        x1 = (-b + math.sqrt(delta)) / (2 * a)
        x2 = (-b - math.sqrt(delta)) / (2 * a)
        print(f"As raízes são: x1 = {x1:.2f} e x2 = {x2:.2f}")

# Executa a função
resolver_segundo_grau()
